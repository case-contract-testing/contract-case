                .request(
                    new HttpRequest(HttpRequestExample.builder()
                        .method("GET")
                        .path("/users/foo")
                        .build())
                )
                .response(new HttpResponse(HttpResponseExample.builder()
                    .status(200)
                    .body(
                        Map.ofEntries(
                            Map.entry("userId", "foo"),
                            Map.entry("name", "john smith")
                        )
                    )
                    .build()))
                .build())
        ),
    );
